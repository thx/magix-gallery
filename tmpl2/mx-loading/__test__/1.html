let Magix = require('magix');
let Loading = require('@../../mx-loading/index');
let $ = require('$');

module.exports = Magix.View.extend({
    tmpl: '@index.html',
    mixins: [Loading],
    render() {
        this.updater.digest();
    },
    'show<click>'(e){
        let that = this;
        that.showLoading();

        setTimeout(() => {
            that.hideLoading();
        }, 3000);
    }
});
