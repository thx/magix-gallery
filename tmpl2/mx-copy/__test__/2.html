以下示例： 复制信息不需要显示，直接配置 copy-text 即可
点此复制链接 {{if success}} 复制成功 {{/if}}
HTML Code
{{!text2}}  
<mx-copy class="btn btn-brand" copy-text="test"
    mx-success="done()">点此复制链接</mx-copy>

{{if success}}
<span class="color-green ml20">复制成功</span>
{{/if}}
JS Code
{{!text3}}  
let Magix = require('magix');
let $ = require('$');

module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        this.updater.digest({
            viewId: this.id
        });
    },
    'done<success>'(e) {
        this.updater.digest({
            success: true
        })
    }
});
