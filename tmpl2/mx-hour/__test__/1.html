let Magix = require('magix');
let Vframe = Magix.Vframe;
let $ = require('$');

module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        this.updater.digest({
            viewId: this.id,
            selected: [{
                week: '12345',
                times: [2, 3, 4]
            }, {
                week: '67',
                times: [10, 11, 12]
            }]
        });
    },
    'get<click>'() {
        let vf = Vframe.get(this.id + '_hour');
        let selected = vf.invoke('val');
        this.updater.digest({
            selected
        })
    }
});
