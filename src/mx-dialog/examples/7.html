let Magix = require('magix');
let Dialog = require('@../index'); //mixins dialog
let $ = require('$');

module.exports = Magix.View.extend({
    tmpl: '@index.html',
    mixins: [Dialog],
    render() {
        this.updater.digest();
    },
    'open<click>'(e) {
        let dlg = this.mxDialog('@./demo');
        dlg.afterClose(() => {
            // do something
        })
    }
});
