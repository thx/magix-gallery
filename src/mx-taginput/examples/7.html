以下示例：
动态刷新列表已选中的传入完整的items:array[object]
用户输入关键词，动态更新提示数据
动态刷的列表只选择一个
请求数据有延时，可加loading
HTML Code
{{!text1}}  
<mx-taginput class="w280"
    dynamic-list="true"
    items="{{@[{
        text: 'one',
        value: 1
    },{
        text: 'two',
        value: 2
    }]}}"
    id="{{=viewId}}_taginput"
    placeholder="请选择需要的值"
    mx-show="update()"/>
JS Code
{{!text2}}  
let Magix = require('magix');
let $ = require('$');
let Vframe = Magix.Vframe;

module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        this.updater.digest();
    },
    'update<show>'(e){
        // e.keyword 用户输入的关键词
        // 用户无输入的情况下不响应
        let keyword = e.keyword;
        if(!keyword){
            return;
        }

        let instance = Vframe.get(this.id + '_taginput');

        // 请求接口有延时，加loading动画
        instance.invoke('showLoading');
        setTimeout(() => {
            let suggest = [1, 2, 3].map(v => {
                return {
                    text: Magix.guid('dynamic_'),
                    value: Magix.guid('dynamic_')
                };
            })

            // 更新下拉框提示数据
            instance.invoke('update', [suggest]);

            // 结束loading
            instance.invoke('hideLoading');
        }, 1500);
    }
});
