mxDialog：
this.mxDialog(viewPath, [viewOptions], [dialogOptions])
默认只做打开浮层的动作，其他什么都不做
可以通过配置header和footer使用内置的吊头和吊尾
当配置了确定按钮时，由于确定按钮在组件内不在浮层区域内，若有确定操作完成之后需要拿到浮层数据，请在 viewOptions 里写 callback 回调
this.mxDialog('@./dialog', {
    callback: (data) => {
        // 回调的浮层数据   
    }
}, {
    width: 400, // 默认400
    height: 400, // 默认无，不设置高度时，自适应由内容撑开
    spm: '', // 用户自定义的spm埋点，不定制则默认生成
    header: { // 默认空，不配置不显示头部
        title: '全屏右出浮层',
        tip: '提示信息',
        iconTip: 'title旁的小问号浮层提示信息'
    },
    footer: {
        enter: '是否需要确认按钮，默认false',
        enterText: '自定义确认按钮文案，默认确定',
        cancel: '是否需要取消按钮，默认false',
        cancelText: '自定义取消按钮文案，默认取消'
    }
});
浮层内：
由于确定按钮在组件内不在浮层区域内，如有确定按钮，请在浮层内实现check方法；
点击确定之前会先调用check，提交按钮加loading效果，避免多次提交；
ok=false的情况下不关闭按钮，如有错误信息显示错误信息，格式如下：
check() {
    // do something

    return new Promise((resolve) => {
        // 此处返回promise，防止有接口提交校验等
        resolve({
            ok: true,  // true or false，是否校验通过，false的时候不关闭按钮
            data: {}, // callback回调数据，可选
            msg: '错误信息提示'
        })
    })
}

let Magix = require('magix');

module.exports = Base.extend({
    tmpl: '@16.html',
    render() {
        this.updater.digest();
    },
    'open1<click>'(e) {
        this.mxDialog('@./dialog', {
            content: '只使用默认头部'
        }, {
            width: 400,
            height: 320, // 指定height的情况下， 组件会计算内容宽度，不指定则自适应
            header: {
                title: '浮层标题',
                tip: '标题旁提示'
            }
        });
    },
    'open2<click>'(e) {
        this.mxDialog('@./dialog', {
            content: '只使用默认尾部'
        }, {
            width: 400,
            height: 320,
            closable: false,
            footer: {
                enter: true,
                cancel: true
            }
        });
    },
    'open3<click>'(e) {
        this.mxDialog('@./dialog', {
            content: '使用默认头部+尾部'
        }, {
            width: 400,
            height: 320,
            header: {
                title: '浮层标题',
                tip: '标题旁提示'
            },
            footer: {
                enter: true,
                cancel: true
            }
        });
    }
});

{{!text5}}  
let Magix = require('magix');

module.exports = Magix.View.extend({
    tmpl: '@dialog.html',
    init(e) {
        this.viewOptions = e;

        this.updater.set({
            content: e.content,
            name: ''
        })
    },
    render() {
        this.updater.digest();
    },
    check() {
        let that = this;
        return new Promise((resolve) => {
            resolve({
                ok: that.isValid(),
                data: {},
                msg: '请按照要求填写信息再提交'
            })
        })
    }
});
