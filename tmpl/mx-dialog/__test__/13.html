规范淘宝登录框的打开样式：
mixins到view上，this.mxLoginView()
有mask，宽度350，高度400（新版mini高度必须大于=400，否则验证码，加验场景展示不全，查看说明 ），居中显示
使用说明：
直接传入bizCode即可，具体逻辑已在组件中包装，目前已有的bizCode定义如下
项目 bizCode 登陆框样式版本 查看示例
阿里妈妈官网 alimamaMain hudongcheng
直通车 subway hudongcheng
钻展 zszw zszwsite_mm
超级推荐 feedFlow dingxiang
超级互动 interactive dingxiang
策略中心 adStrategy dingxiang
达摩盘服务商版本 dmpIsv mdmp
联盟媒体 unionMedia hudongcheng
联盟商家 unionMerchant hudongcheng
品销宝 brand dingxiang
tanx opensspTanx dingxiang
默认 dingxiang

JS Code
{{!text2}}  
let Magix = require('magix');
let Dialog = require('@../index'); //mixins dialog

module.exports = Magix.View.extend({
    tmpl: '@index.html',
    mixins: [Dialog],
    render() {
        this.updater.digest();
    },
    'show<click>'(e) {
        // this.mxLoginView(viewOptions)
        // viewOptions：object
        //      bizCode：项目bizCode，包装了登陆框逻辑
        this.mxLoginView({
            bizCode: 'adStrategy'
        })
    }
});
