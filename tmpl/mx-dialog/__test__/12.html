以下示例：
指定target默认相对节点下居中打开浮层
当前节点下居中 当前节点下居中 + 偏移
HTML Code
{{!text1}}  
<a href="javascript:;" class="btn btn-brand mr20"
    id="{{=viewId}}_test"
    mx-click="open1()">当前节点下居中</a>

<a href="javascript:;" class="btn btn-brand mr20"
    mx-click="open2()">当前节点下居中 + 偏移</a>
HTML Code
{{!text2}}  
let Magix = require('magix');
let $ = require('$');

module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        this.updater.digest();
    },
    'open1<click>'(event) {
        this.mxDialog('@./demo', {}, {
            target: `#${this.id}_test` 
        });
    },
    'open2<click>'(event) {
        this.mxDialog('@./demo', {}, {
            target: $(event.target),
            offset: {
                left: 100,
                top: -20
            }
        });
    }
});
