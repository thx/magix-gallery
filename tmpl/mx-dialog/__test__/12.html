以下示例：
指定target默认相对节点下居中打开浮层
HTML Code
{{!text1}}  
<mx-btn id="{{=viewId}}_test"
    mode="primary" 
    content="当前节点下居中"
    mx-click="open1()"/>

<mx-btn mode="primary" 
    content="当前节点下居中 + 偏移"
    mx-click="open2()"/>
JS Code
{{!text2}}  
let Magix = require('magix');

module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        this.updater.digest();
    },
    'open1<click>'(event) {
        this.mxDialog('@./demo', {}, {
            target: `#${this.id}_test` 
        });
    },
    'open2<click>'(event) {
        this.mxDialog('@./demo', {}, {
            target: $(event.target),
            offset: {
                left: 100,
                top: -20
            }
        });
    }
});
