全屏右出浮层：
this.mxModal(viewPath, [viewOptions], [dialogOptions])
由于确定按钮在组件内不在浮层区域内，如有确定操作完成之后需要拿到浮层数据，请在 viewOptions 里写 callback 回调
dialog-header和dialog-footer都在组件内，配置即可，集中关心中间区域即可；
默认mask，宽度600
this.mxModal('@./full', {
    callback: (data) => {
        // 回调的浮层数据   
    }
}, {
    width: 800,
    header: {
        title: '全屏右出浮层',
        tip: '提示信息'
    },
    footer: {
        enter: '是否需要确认按钮',
        enterText: '自定义确认',
        cancel: '是否需要取消按钮',
        cancelText: '自定义取消'
    }
});
浮层内：
由于确定按钮在组件内不在浮层区域内，如有确定按钮，请在浮层内实现check方法；
点击确定之前会先调用check，提交按钮加loading效果，避免多次提交；
ok=false的情况下不关闭按钮，如有错误信息显示错误信息，格式如下：
check() {
    // do something

    return new Promise((resolve) => {
        // 此处返回promise，防止有接口提交校验等
        resolve({
            ok: true,  // true or false，是否校验通过，false的时候不关闭按钮
            data: {}, // callback回调数据，可选
            msg: '错误信息提示'
        })
    })
}

let Magix = require('magix');

module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        this.updater.digest();
    },
    check() {
        let that = this;
        let ok = that.isValid();
        let name = '';
        if(ok){
            name = that.updater.get('name');
        }
        return new Promise((resolve) => {
            setTimeout(() => {
                resolve({
                    ok,
                    data: {
                        name
                    },
                    msg: '请按照要求填写完信息再提交'
                })
            }, 1000)
        })
    }
}, {
    // 浮层的默认宽度
    dialogOptions: {
        width: 1000
    }
});
