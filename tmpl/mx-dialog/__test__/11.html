afterClose：
浮层关闭之后调用
包括取消按钮关闭 + 右上角关闭 + 点击空白处关闭，任意关闭浮层的操作都会触发afterClose
HTML Code
{{!text1}}  
<mx-btn mode="primary" 
    content="打开浮层"
    mx-click="open()"/>
JS Code
{{!text2}}  
let Magix = require('magix');
module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        this.updater.digest();
    },
    'open<click>'(e) {
        let dlg = this.mxDialog('@./demo');
        dlg.afterClose(() => {
            // do something
        })
    }
});
