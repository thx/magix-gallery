let Magix = require('magix');

module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        this.updater.digest();
    },
    check() {
        let that = this;
        let ok = that.isValid();
        let name = '';
        if(ok){
            name = that.updater.get('name');
        }
        return new Promise((resolve) => {
            setTimeout(() => {
                resolve({
                    ok,
                    data: {
                        name
                    },
                    msg: '请按照要求填写完信息再提交'
                })
            }, 1000)
        })
    }
}, {
    dialogOptions: {
        width: 1000
    }
});
