展示优先级：
某些场景下，浮层展示尺寸是固定的，则可在浮层内配置固定展示属性
外部自定义属性 > 浮层内属性 > 默认属性
浮层本身写法
{{!text3}}  
let Magix = require('magix');

module.exports = Magix.View.extend({
    tmpl: '@inner.html',
    init(e) {
        this.viewOptions = e;
    },
    render() {
        this.updater.digest();
    },
}, {
    dialogOptions: {
        top: 0,
        width: 800,
        header: {
            title: '浮层标题'
        },
        footer: {
            enter: false,
            cancel: true,
        }
    }
});
使用浮层内配置的属性
{{!text1}}  
'open<click>'(e) {
    this.mxDialog('@./inner');
}
自定义属性
{{!text2}}  
'open<click>'(e) {
    this.mxDialog('@./inner', {}, {
        top: 100,
        width: 400
    });
}
