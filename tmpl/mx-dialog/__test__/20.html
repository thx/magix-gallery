以下示例：
支持配置footer.view & footer.viewData 指定按钮旁自定义view
footer.view需要实现check方法，若存在校验，提交时会一并处理
HTML Code
{{!text1}}  
import Magix, { Vframe } from 'magix';

export default Magix.View.extend({
    tmpl: '@index.html',
    render() {
        this.updater.digest();
    },
    'open<click>'(e) {
        this.mxModal('@./modal-inner', {
            enterCallback: (result) => {
                this.alert('系统提示', `按钮旁输入内容为“${result.footerName}”`);
            }
        }, {
            ladder: {
                width: 'xlarge',
            },
            header: {
                title: '浮层标题',
                tip: '按钮旁支持配置footer.view配置自定义嵌入view',
            },
            footer: {
                view: '@./dialog-footer',
                viewData: {
                    footerName: '文案'
                }
            }
        });
    },
});
dialog-footer HTML Code
{{!text2}}  
import Magix from 'magix';

export default Magix.View.extend({
    tmpl: '@dialog-footer.html',
    init(e) {
        this.viewOptions = e;

        this.updater.set({
            footerName: e.data?.footerName || ''
        })
    },
    render() {
        this.updater.digest();
    },
    check() {
        let that = this;
        return new Promise((resolve) => {
            resolve({
                ok: that.isValid(),
                data: {
                    footerName: this.updater.get('footerName'),
                },
                msg: '请按照要求填写信息再提交'
            })
        })
    }

});
