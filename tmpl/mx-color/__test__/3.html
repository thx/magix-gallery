let Magix = require('magix');
let $ = require('$');

module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        this.updater.digest({
            color: '#9a3c3c'
        });
    },
    'changeColor<change>'(e){
        this.updater.digest({
            color: e.color
        });
    }
});
