let Magix = require('magix');
module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        let list = [{
            value: 1,
            text: '模块1'
        }, {
            value: 2,
            text: '模块2'
        }, {
            value: 3,
            text: '模块3'
        }];

        this.updater.digest({
            list,
            selected: list[0].value
        });
    },

    'changeTab<change>' (e) {
        // e.selected 当前选中的key值
        // e.value 当前选中的key值
        // e.text 当前选中的文案
        this.updater.digest({
            selected: e.selected
        })
    }
});
