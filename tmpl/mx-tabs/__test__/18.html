let Magix = require('magix');
module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        let list = [];
        for (let i = 1; i < 4; i++) {
            list.push({
                value: i,
                text: `模块${i}`,
            })
        }

        this.updater.digest({
            list,
            selected: list[0].value
        });
    },

    'changeTab<change>'(e) {
        // e.selected 当前选中的key值
        // e.value 当前选中的key值
        // e.text 当前选中的文案
        this.updater.digest({
            selected: e.selected
        })
    }
});
