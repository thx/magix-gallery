打标支持： 代码片段或者纯文字，支持自定义背景颜色，默认品牌色，禁用状态下为灰色
HTML Code
{{!text1}}  
<!-- 打标优先级 tagContent > tag -->
<mx-tabs.box  list="{{@list}}"/>

<!-- 自定义打标颜色 -->
<mx-tabs.box    
    list="[{
        text: '模块1',
        value: 1,
        tag: 'New',
        color: '#4d7fff'
    },{
        text: '模块2',
        value: 2
    }]"/>

<!-- 禁用 -->
<mx-tabs.box
    disabled="true"
    list="[{
        text: '模块1',
        value: 1,
        tag: 'New'
    },{
        text: '模块2',
        value: 2
    }]"/>
JS Code
{{!text2}}  
let Magix = require('magix');

module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        this.updater.digest({
            list: [{
                text: '模块1',
                value: 1,
                tag: '<i class="mc-iconfont">&#xe601;</i>',
                tips: '提示信息',
            }, {
                text: '模块2',
                value: 2,
                tagContent: '<span style="position: absolute; left: 0; top: -6px; width: 8px; height: 8px; border-radius: 50%; background-color: var(--color-red);"></span>',
            }],
        });
    }
});
