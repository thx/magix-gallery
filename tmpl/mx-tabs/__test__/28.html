展示完整的二级菜单导航示例
有二级菜单只有非禁用状态下展开，前序数值配置finished=true显示为打钩，配置highlight=true前序数值高亮为品牌色
当前选中：{{=selected}}
HTML Code
{{!text1}}  
<mx-tabs.pipeline 
    mode="nav"
    selected="{{=selected}}"
    list="{{@list}}"
    mx-change="change()"/>
JS Code
{{!text2}}  
let Magix = require('magix');
module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        let list = [{
            value: 1,
            text: '一级菜单1',
            finished: true,
            icon: 'https://img.alicdn.com/tfs/TB13e.TioY1gK0jSZFCXXcwqXXa-28-34.png'
        }, {
            value: 2,
            text: '一级菜单2',
            finished: true,
            subs: [{
                value: 21,
                text: '二级菜单2_1'
            }, {
                value: 22,
                text: '二级菜单2_2'
            }]
        }, {
            value: 3,
            text: '一级菜单3',
            highlight: true,
            iconTip: '提示信息',
        }, {
            value: 4,
            text: '一级菜单4',
            disabled: true
        }, {
            value: 5,
            text: '一级菜单5',
            disabled: true
        }]

        this.updater.digest({
            viewId: this.id,
            list,
            selected: list[2].value,
        });
    },
    'change<change>'(e) {
        this.updater.digest({
            selected: e.selected,
        })
    }
});
