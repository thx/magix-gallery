以下示例：
切换tab前有同步校验逻辑，当输入一个非整数时不允许切换
该方式仅适用于同步校验场景
菜单{{=selected}}
HTML Code
{{!text1}}  
<mx-tabs 
    list="{{@list}}" 
    selected="{{=selected}}"
    mx-change="changeTab()" />
    
<mx-grid.body>
    <form class="form">
        <div class="form-item">
            <div class="form-label">菜单{{=selected}}</div>
            <div class="form-content">
                <mx-input class="w200" placeholder="请输入整数" value="{{:values[selected]{int:true}}}" />
            </div>
        </div>
    </form>
</mx-grid.body>
JS Code
{{!text2}}  
let Magix = require('magix');

module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        let list = [{
            value: 1,
            text: '菜单1',
        }, {
            value: 2,
            text: '菜单2',
        }, {
            value: 3,
            text: '菜单3',
        }];

        let values = {};
        list.forEach(item => {
            values[item.value] = '';
        })

        this.updater.digest({
            list,
            selected: list[0].value,
            values
        });
    },

    'changeTab<change>'(e) {
        let that = this;
        if (that.isValid()) {
            that.updater.digest({
                selected: e.value
            })
        } else {
            // preventDefault 阻断切换
            e.preventDefault();
        }
    },
});
