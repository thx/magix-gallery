let Magix = require('magix');
module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        let viewId = this.id;

        this.updater.digest({
            viewId,
            list: [{
                value: `${viewId}_1`,
                text: '红色模块',
                tag: 'New'
            }, {
                value: `${viewId}_2`,
                text: '绿色模块',
                tag: 'New',
                color: '#30ab66'
            }, {
                value: `${viewId}_3`,
                text: '黄色模块',
                tip: '提示信息'
            }]
        });
    }
});
