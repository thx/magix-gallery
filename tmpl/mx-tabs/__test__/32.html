展示完整的二级菜单导航示例
每次操作仅展开一个二级菜单，支持配置expand=false表示收起状态
支持展开收起，收起时点击icon选中子选项中第一个选项
当前选中：{{=selected}}
HTML Code
{{!text1}}  
<mx-tabs.pipeline 
    mode="menu"
    selected="{{=selected}}"
    list="{{@list}}"
    mx-change="change()"/>
JS Code
{{!text2}}  
let Magix = require('magix');
module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        let list = [{
            value: 1,
            text: '一级菜单1',
            icon: '<i class="mc-iconfont">&#xe91d;</i>', // 菜单前缀icon
        }, {
            value: 2,
            text: '一级菜单2',
            icon: '<i class="mc-iconfont">&#xe671;</i>',
            tagContent: '<img src="https://img.alicdn.com/tfs/TB13e.TioY1gK0jSZFCXXcwqXXa-28-34.png" />', // 打标html片段
            subs: [{
                value: 21,
                text: '二级菜单2_1',
            }, {
                value: 22,
                text: '二级菜单2_2',
            }]
        }, {
            value: 3,
            text: '一级菜单3',
            icon: '<i class="mc-iconfont">&#xe780;</i>',
            subs: [{
                value: 31,
                text: '二级菜单3_1',
                tag: 'HOT', // 打标文案
            }, {
                value: 32,
                text: '二级菜单3_2'
            }]
        }, {
            value: 4,
            text: '一级菜单4',
            icon: '<i class="mc-iconfont">&#xe60e;</i>',
        }]

        this.updater.digest({
            viewId: this.id,
            list,
            selected: 21,
        });
    },
    'change<change>'(e) {
        this.updater.digest({
            selected: e.selected,
        })
    }
});
