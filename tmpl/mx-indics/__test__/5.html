当前操作： {{=triggerType}}
指标类型： {{if custom}} 自定义指标 {{else}} 默认指标 {{/if}}
选中指标： {{if custom}} {{each customs as key}} {{=map[key]}}（{{=key}}） {{/each}} {{else}} {{each defaults as key}} {{=map[key]}}（{{=key}}） {{/each}} {{/if}}

let Magix = require('magix');
module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        let fields = [{
            value: 1, text: '消耗', pValue: 1, tip: '消耗提示文案'
        }, {
            value: 2, text: '展现量', pValue: 1
        }, {
            value: 3, text: '点击量', pValue: 1
        }, {
            value: 4, text: '点击率', pValue: 1
        }, {
            value: 5, text: '平均点击单价', pValue: 1, tip: '平均点击单价提示文案'
        }, {
            value: 6, text: '点击转化率', pValue: 2
        }, {
            value: 7, text: '成交笔数', pValue: 2
        }, {
            value: 8, text: '千次展现成本', pValue: 2
        }, {
            value: 9, text: '投资回报率', pValue: 2
        }, {
            value: 10, text: '访问页面数', pValue: 2
        }]

        let parents = [{
            value: 1, text: '组1'
        }, {
            value: 2, text: '组2'
        }]
        
        this.updater.digest({
            parents,
            fields,
            custom: false, //当前是为默认数据
            customs: [2,3,4,5],
            defaults: [2,3]
        });
    },
    'change<change>'(e){
        // e.custom：true or false，是否为自定义指标
        // e.customs：用户自定义的指标，当前用的的配置
        // e.defaults：默认指标，不变
        // e.triggerType：操作类型
        //      btn-switch 切换自定义和默认设置
        //      dialog-setting 浮层编辑指标
    }
});
