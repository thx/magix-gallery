鼠标hover节点自定义更新left + top
HTML Code

    content="自定义left + top"
    top="{{=top}}"
    left="{{=left}}"
    auto="{{=auto}}"
    mx-mouseover="enter()">鼠标hover节点自定义更新left + top</mx-popover>
JS Code
{{!text2}}  
let Magix = require('magix');
module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        this.updater.digest({
            auto: false, //控制节点展开
            top: '',
            left: ''
        });
    },
    'enter<mouseover>' (event) {
        if (Magix.inside(event.relatedTarget, event.eventTarget)) {
            return;
        }

        let node = $(event.eventTarget);
        let offset = node.offset();

        this.updater.digest({
            auto: true,
            top: offset.top,
            left: offset.left + 220
        })
    }
});
