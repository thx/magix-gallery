响应change事件

HTML Code:
<mx-view path="mx-slider/index"
    max="200"
    min="100"
    value="130"
    step="5"
    mx-change="showValue()"
    class="fl"
></mx-view>
Javascript Code:
let Magix = require('magix');
let GTip = require('app/gallery/mx-gtip/index');
module.exports = Magix.View.extend({
    tmpl: '@index.html',
    mixins: [GTip],
    render() {
        let me = this;
        me.updater.digest();
    },
    'showValue<change>' (e) {
        this.gtipRT('value:' + e.value);
    }
});
