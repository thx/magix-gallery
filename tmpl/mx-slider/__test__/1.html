包含输入框
输入框与滑动轴联动
step几位小数，其他数值同step保持一致
配置tip带提示文案
配置width自定义宽度，默认280
当前值： {{=cur}}
HTML Code

<mx-slider
    width="160"
    need-input="true"
    max="200"
    min="100"
    step="0.05"
    tip="元"
    value="{{=cur}}"
    mx-change="showValue()"/>
JS Code
{{!text2}}  
let Magix = require('magix');

module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        this.updater.digest({
            cur: 120
        });
    },
    'showValue<change>' (e) {
        // 操作结束的时候返回
        // 当前值 e.value
        this.updater.digest({
            cur: e.value
        })
    }
});
