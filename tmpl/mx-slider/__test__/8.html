开始值： {{=start}} 结束值： {{=end}}
HTML Code
{{!text1}}  
<mx-slider.range
    max="200"
    min="100"
    selected="{{=start}},{{=end}}"
    step="0.05"
    tip="元"
    mx-change="change()"/>
JS Code
{{!text2}}  
let Magix = require('magix');

module.exports = Magix.View.extend({
    tmpl: '@8.html',
    render() {
        this.updater.digest({
            start: 110,
            end: 150
        });
    },
    'change<change>'(e) {
        // 操作结束的时候返回
        // e.selected [array / string] [min, max] 或者 min,max 与入参一致
        // e.start 开始值
        // e.end 结束值
        this.updater.digest({
            start: e.start,
            end: e.end
        })
    }
});
