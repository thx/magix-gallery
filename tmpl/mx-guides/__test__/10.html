let Magix = require('magix');
let Guides = require('@../../mx-guides/index');

module.exports = Magix.View.extend({
    tmpl: '@index.html',
    mixins: [Guides],
    render() {
        this.updater.digest({
            viewId: this.id
        });
    },
    'show<click>'(e) {
        this.showMxGuides({
            mode: 'point',
            type: 'breath',
            list: [{
                sizzle: `#${viewId}_btn`,
                tip: '功能上新啦！',
                placement,
            }]
        });
    }
});
