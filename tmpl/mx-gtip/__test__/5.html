JS Code
{{!text2}}  
let Magix = require('magix');
let GTip = require('@../../mx-gtip/message');

module.exports = Magix.View.extend({
    tmpl: '@index.html',
    mixins: [GTip],
    render() {
        this.updater.digest();
    },
    /**
     * 错误提示
     */
    'show1<click>'(e) {
        // 当前数据config会完整传递到子view
        // 如果有额外参数可直接在config中增加
        this.gmessage({
            type: 'error',
            view: '@./custom',
            something: {}, // 额外信息
        });
    },
    /**
     * 警告提示
     */
    'show2<click>'(e) {
        this.gmessage({
            type: 'warn',
            msg: '警告类提示信息，4s自动关闭',
            timeout: 4000
        });
    },
    /**
     * 品牌色提示（默认设置）
     */
    'show3<click>'(e) {
        this.gmessage({
            type: 'highlight',
            msg: '品牌色提示信息，默认提示样式',
        });
    },
    /**
     * 通过提示
     */
    'show4<click>'(e) {
        this.gmessage({
            type: 'pass',
            msg: '提示信息，5s自动关闭',
            timeout: 5000
        });
    },
    /**
     * 黑色透明度底
     */
    'show5<click>'(e) {
        this.gmessage({
            type: 'common',
            msg: '黑色透明度底提示信息'
        });
    }
});
