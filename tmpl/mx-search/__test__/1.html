支持上下键切换选择项
未搜索状态下，回车默认搜索第一个
已搜索状态下，回车搜索当前选中项
默认placeholder=list.map(item => {return item.text;})，对item.text去重
事件处理
当前搜索： {{if searchKey}} {{=searchKey}} = {{=searchValue}} {{/if}}

    list="{{@list}}" 
    search-key="{{=searchKey}}"

    mx-search="change()"></mx-search>
JS Code
{{!text2}}  
let Magix = require('magix');
let $ = require('$');

module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        this.updater.digest({
            list: [{
                text: '计划',
                value: 'campaignId'
            }, {
                text: '单元',
                value: 'adgroupId'
            }],
            searchKey: 'adgroupId',
            searchValue: '12'
        });
    },
    'change<search>' (e) {
        e.preventDefault();
        // e.searchKey 搜索对应的key值
        // e.searchValue input文本框输入的对应的内容
        this.updater.digest({
            searchKey: e.searchKey,
            searchValue: e.searchValue
        })
    }
});
