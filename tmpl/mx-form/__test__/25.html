inranges：
命中配置区间则异常，配置区间之外的通过；() 表示不包含边界值，[] 表示包含边界值；
inranges: [
    ['(,2)', '<2'],
    ['[2, 3]', '2 <= x <= 3'],
    ['(3, 5]', '3 < x <= 5'],
    ['(5,)', '>5']
]

JS Code
{{!text2}}  
let Magix = require('magix');
module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        let errorTip = '输入数字x，必须 3 <= x < 5 或者 x >= 9',
            warnTip = '输入数字x，建议 x < 3 或者 9 <= x < 10';
        this.updater.digest({
            errorTip,
            errorRules: {
                number: true,
                inranges: [
                    ['(,3)', errorTip],
                    ['[5, 9)', errorTip],
                ]
            },
            warnTip,
            warnRules: {
                number: true,
                warn: {
                    inranges: [
                        ['[3, 9)', warnTip],
                        ['[10,)', warnTip],
                    ]
                }
            }
        });
    }
});
