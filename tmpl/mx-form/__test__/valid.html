方法名 说明
isValid(configs: object) 
this.isValid({
        // 选择器
        // 1. 默认null，整个view校验
        // 2. 传入element的时候
        //    2.1 如果element本身有校验，则只校验该节点
        //    2.2 如果element本身无校验，则找该节点下所有的校验项去校验
        sizzle: null,
    
        // 校验有错误的情况下是否要滚动到错误节点，默认true
        scrollIntoView: true, 
    
        //是否调用子view校验，children.isValid，默认true
        checkSubs: true 
})
