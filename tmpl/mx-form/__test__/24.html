使用说明：
style=text，纯文案提示
配置placement可设置提示文案位置，可选right，bottom，默认bottom
错误文案
警告文案
成功提醒文案
品牌色强调文案
HTML Code
{{!text1}}  
<!-- 错误文案 -->
<mx-input class="w200" placeholder="下方必填" value="{{:name{required:true}}}"/>
<mx-input class="w200" placeholder="右侧必填" value="{{:name{required:true,placement:'right'}}}"/>
    
<!-- 警告文案 -->
<mx-input class="w200" placeholder="下方必填小于5警告" value="{{:name{required:true,warn:{min: [5, '小于5给警告提示']}}}}"/>
<mx-input class="w200" placeholder="右侧必填小于5警告" value="{{:name{required:true,placement:'right',warn:{placement:'right',min: [5, '小于5给警告提示']}}}}"/>

<!-- 成功提醒文案 -->
<mx-input class="w200" placeholder="下方必填小于5表示成功" value="{{:name{required:true,pass:{min: [5, '小于5给成功提示']}}}}"/>
<mx-input class="w200" placeholder="右侧必填小于5表示成功" value="{{:name{required:true,placement:'right',pass:{placement:'right',min: [5, '小于5给成功提示']}}}}"/>

<!-- 品牌色强调文案 -->
<mx-input class="w200" placeholder="下方必填小于5品牌色高亮" value="{{:name{required:true,highlight:{min: [5, '小于5给品牌色高亮提示']}}}}"/>
<mx-input class="w200" placeholder="右侧必填小于5品牌色高亮" value="{{:name{required:true,placement:'right',highlight:{placement:'right',min: [5, '小于5给品牌色高亮提示']}}}}"/>
