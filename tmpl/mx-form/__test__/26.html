HTML Code
{{!text1}}  
<!-- 错误文案 -->
<mx-input class="w200" placeholder="下方必填" 
    value="{{:name{required:true,style:'box'}}}"/>

<!-- 警告文案 -->
<mx-input class="w200" placeholder="下方必填小于5警告" 
    value="{{:name{required:true,style:'box',warn:{style:'box',min: [5, '小于5给警告提示']}}}}"/>

<!-- 成功提醒文案 -->
<mx-input class="w200" placeholder="下方必填小于5表示成功" 
    value="{{:name{required:true,style:'box',pass:{style:'box',min: [5, '小于5给成功提示']}}}}"/>

<!-- 品牌色强调文案 -->
<mx-input class="w200" placeholder="下方必填小于5品牌色高亮" 
    value="{{:name{required:true,style:'box',highlight:{style:'box',min: [5, '小于5给品牌色高亮提示']}}}}"/>
