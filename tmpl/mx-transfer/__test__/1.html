import Magix from 'magix';
export default Magix.View.extend({
    tmpl: '@index.html',
    render() {
        let list = [];
        for (let i = 1; i < 11; i++) {
            list.push({
                text: `选项${i}`,
                value: i,
                tag: (i == 2) ? 'NEW' : '',
            })
        }

        this.updater.digest({
            list,
            selected: []
        });
    },
    'change<change>'(e) {
        // e.selected：当前选中值，初始化为什么类型就保持什么类型，默认string
        // e.items：当前选中值完整对象数组
        this.updater.digest({
            selected: e.selected,
        })
    }
});
