以下示例：
部分文案可编辑，配置展示模板
hover元素显示编辑按钮
回车和失去焦点都可触发回调，只触发一次
当前内容： {{=content}}（要求必填且不小于200）
HTML Code
{{!text1}}  
<mx-editor
    tmpl="<span>${content}</span><span class='grey-solid-icon ml5'>打标</span>"
    content="{{=content}}"
    rules="{{@rules}}"
    mx-change="change()"></mx-editor>
JS Code
{{!text2}}  
let Magix = require('magix');
let $ = require('$');

module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        this.updater.digest({
            content: 2000,
            rules: {
                required: true,
                min: [200, '不小于200']
            }
        });
    },
    'change<edit>' (e) {
        // editText编辑后的文案
        this.updater.digest({
            content: e.editText
        })
    }
});
