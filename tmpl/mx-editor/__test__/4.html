以下示例： 传入格式化函数replace-fn，自行处理显示样式
当前输入内容： {{=content}}
HTML Code
{{!text1}}  
<mx-editor
    replace-fn="{{@replaceFn}}"
    content="{{:content{refresh:true}}}"
    rules="{{@{
        required: true,
        min: [200, '不小于200']
    }}}"/>
JS Code
{{!text2}}  
let Magix = require('magix');
module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        this.updater.digest({
            content: 200.22,
            replaceFn: (str) => {
                let arr = ((str || '') + '').split('.');
                return `<span class="fontsize-20 bold">${arr[0]}</span>${arr[1] ? ('<span class="fontsize-14 bold">.' + arr[1] + '</span>') : ''}<img class="ml4" width="12" src="https://img.alicdn.com/imgextra/i2/O1CN014Tqfav1kV9WwsKZzJ_!!6000000004688-2-tps-24-28.png" />`;
            }
        });
    }
});
