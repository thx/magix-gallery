以下示例：
最多显示20个字符，超出则...缩略显示
当前内容：
{{=content}}
实际显示：
HTML Code
{{!text1}}  
<mx-editor class="w140"
    tmpl="{{=tmpl}}"
    content="{{=content}}"
    rules="{{@{
        required: true
    }}}"
    mx-change="change()" />
JS Code
{{!text2}}  
let Magix = require('magix');

module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        let content = '这是一条非常长非常长非常长的需要换行的数据';
        this.setContent(content);
    },
    'change<edit>' (e) {
        // editText编辑后的文案
        this.setContent(e.editText);
    },
    setContent(content){
        let len = content.length,
            max = 20;
        this.updater.digest({
            tmpl: (len > max) ? (content.slice(0, max) + '...') : content,
            content
        });
    }
});
