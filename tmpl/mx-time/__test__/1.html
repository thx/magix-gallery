下列三种写法等价，时分秒均可选：
types=""
types="all"
types="hour,minute,second"
当前时间： {{=time}}
HTML Code

<mx-time class="w120" 
    id="{{=viewId}}_time"
    time="10:10:10"
    mx-change="change()" />

<!-- 等价于 -->
<mx-time class="w120" 
    id="{{=viewId}}_time"
    time="10:10:10"
    types="all"
    mx-change="change()" />

<!-- 等价于 -->
<mx-time class="w120" 
    id="{{=viewId}}_time"
    time="10:10:10"
    types="hour,minute,second"
    mx-change="change()" />
