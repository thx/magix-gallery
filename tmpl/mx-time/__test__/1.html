下列三种写法等价，时分秒均可选：
  • types=""
  • types="all"
  • types="hour,minute,second"
当前时间：{{=time}}
HTML Code
{{!text1}}  
<mx-time class="w120" 
    time="10:10:10"
    mx-change="change()" />

<!-- 等价于 -->
<mx-time class="w120" 
    time="10:10:10"
    types="all"
    mx-change="change()" />

<!-- 等价于 -->
<mx-time class="w120" 
    time="10:10:10"
    types="hour,minute,second"
    mx-change="change()" />
JS Code
{{!text2}}  
let Magix = require('magix');
module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        this.updater.digest({
            time: '10:10:10'
        });
    },
    'change<change>'(e) {
        // e.time：当前时间
        this.updater.digest({
            time: e.time,
        })
    }
});
