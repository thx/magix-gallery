方法名 说明 调用方法
val() 忽略校验，直接获取选中结果 
let instance = Vframe.get(id);

// selected = [{
//     week: 1, //周一
//     times: [2,3,4]
// },
// {
//     week: 2, //周二
//     times: [2,3,4]
// },
// ...
// {
//     week: 7, //周日
//     times: [1,2,4]
// }]
let selected = instance.invoke('val');

let instance = Vframe.get(id);

//      },
//      ...
//      {
//          week: 7, //周日
//          times: [1,2,4]
//      }]
// }
let result = instance.invoke('check');
