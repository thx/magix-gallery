以下示例： 直接使用日历面板
当前日期： {{=configs.selected}}
HTML Code
{{!text1}}  
<div class="lh28 mb20">
    <span class="color-9">当前日期：</span>
    <span>{{=configs.selected}}</span>
</div>
<mx-calendar
    configs="{{@configs}}"
    mx-change="change()"/>
JS Code
{{!text2}}  
let Magix = require('magix');

module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        this.updater.digest({
            configs: {
                min: '{{=configs.min}}',
                max: '{{=configs.max}}',
                selected: '{{=configs.selected}}',
            }
        });
    },
    'change<change>'(e){
        // e.selected
        let configs = this.updater.get('configs');
        this.updater.digest({
            configs: Magix.mix(configs, {
                selected: e.selected,
            })
        })
    }
});
