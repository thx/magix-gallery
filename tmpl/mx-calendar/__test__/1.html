JS Code
{{!text2}}  
let Magix = require('magix');
let $ = require('$');

module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        this.updater.digest({
            min: '{{=min}}',
            max: '{{=max}}',
            selected: '{{=selected}}'
        });
    },

    'changeDate<click>'(event){
        this.updater.digest({
            selected: '{{=today}}'
        })
    },
    
    'select<change>' (event) {
        // event.date 当前选择日期
        // event.time 时分秒
        this.updater.digest({
            selected: event.date
        })
    }
});
