以下示例： 可对比，关闭对比选择单天
当前结果： {{if vs}} {{=start}} 对比 {{=end}} {{else}} {{=start}} {{/if}}
HTML Code
{{!text1}}  
<mx-calendar.rangepicker class="w220"
    vsenable="true"
    vs="{{=vs}}"
    single="true"
    start="{{=start}}"
    end="{{=end}}"
    max="{{=max}}"
    min="{{=min}}"
    mx-change="changeInfo()"/>
JS Code
{{!text2}}  
let Magix = require('magix');

module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        this.updater.digest({
            vsenable: true,
            vs: true,
            start: '{{=start}}',
            end: '{{=end}}',
            min: '{{=min}}',
            max: '{{=max}}',
        });
    },
    'changeInfo<change>'(event) {
        let data = this.updater.get();
        // event.dates.startStr：开始日期
        // event.dates.endStr：结束日期，不限时未文案不限
        // event.vs：是否对比
        let dates = event.dates,
            vs = event.vs;
        this.updater.digest({
            vs: vs,
            start: dates.startStr,
            end: dates.endStr
        });
    },
});
