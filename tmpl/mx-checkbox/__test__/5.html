let Magix = require('magix');
let $ = require('$');
module.exports = Magix.View.extend({
    tmpl: '@1.html',
    render() {
        let list = [{
            text: '选项文案1',
            value: 1,
            tip: '选项说明文案一行或者两行高度一致',
            icon: '<i class="mc-iconfont"></i>',
        }, {
            text: '选项文案2',
            value: 2,
            tip: '选项说明文案一行或者两行高度一致',
        }, {
            text: '选项文案3',
            value: 3,
            tip: '选项说明文案一行或者两行高度一致',
            icon: '<i class="mc-iconfont"></i>',
        }];

        this.updater.digest({
            list,
            selected: [list[0].value]
        });
    },

    'change<change>'(e) {
        // e.selected 当前选中value值
        // 传入数组为数组，传入逗号分隔为逗号分隔，默认逗号分隔
        this.updater.digest({
            selected: e.selected
        })
    }
});
