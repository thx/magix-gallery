使用说明：
mx-checkbox同原生checkbox一样支持双向绑定
以下mx-checkbox必选
单个布尔值： 当前选中：{{=selected7}}
数组（至少一个）： {{each list as item}} {{/each}} 当前选中：{{=selected1}}
HTML Code
{{!text1}}  
<span>单个布尔值：</span>
<mx-checkbox class="mr30"
    name="{{=viewId}}_checkboxs_1" 
    value="true" 
    text="是否选中为布尔值"
    {{:selected1{refresh:true}}} />

<span>数组（至少一个）：</span>
{{each list as item}}
<mx-checkbox class="mr30"
    name="{{=viewId}}_checkboxs_1" 
    value="{{=item.value}}" 
    text="{{=item.text}}"
    {{:selected2{required:[true,'必选'],refresh:true}}} />
{{/each}}
JS Code
{{!text2}}  
let Magix = require('magix');

module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        let list = [];
        for (let i = 1; i < 5; i++) {
            list.push({
                value: i,
                text: `选项${i}`,
                tip: '提示信息',
                tag: '打标',
            })
        }

        this.updater.digest({
            viewId: this.id,
            list,
            selected1: true,
            selected2: [1],
        });
    }
});
