以下示例：
切换开关前有同步校验逻辑，使用e.preventDefault阻断切换
该方式仅适用于同步校验场景
HTML Code
{{!text1}}  
<mx-switch 
    state="{{=state}}" 
    loading="{{=loading}}"
    mx-change="change()"/>
JS Code
{{!text2}}  
let Magix = require('magix');
module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        this.updater.digest({
            state: false,
            loading: false
        });
    },
    'change<change>'(e) {
        // 不切换
        e.preventDefault();
    }
});
