以下示例：两种模式的开关切换样式
当前选中：{{=selected}}
HTML Code
{{!text1}}  
<mx-switch.tab 
    list="{{@list}}"
    selected="{{=selected}}"
    mx-change="change()" />
JS Code
{{!text2}}  
import Magix from 'magix';
export default Magix.View.extend({
    tmpl: '@index.html',
    render() {
        let list = [{
            text: '极简版',
            value: 'simple',
            icon: '<i class="mx-iconfont">&#xe64d;</i>'
        }, {
            text: '专业版',
            value: 'custom',
            icon: '<i class="mx-iconfont">&#xe65f;</i>',
        }];

        this.updater.digest({
            list,
            selected: list[0].value,
        })
    },
    'change<change>'(e) {
        this.updater.digest({
            selected: e.selected,
        })
    }
});
