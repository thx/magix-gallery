以下示例： 某些场景下，开关并不是boolean值，而是一个字符串枚举，可配置valueMap:{true: 1, false: 2}进行快捷映射
当前选中：{{=state}}
HTML Code
{{!text1}}  
<mx-switch 
    state="{{:state{refresh:true}}}" 
    value-map="{{@valueMap}}"/>
JS Code
{{!text2}}  
let Magix = require('magix');
module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        this.updater.digest({
            state: 1,
            valueMap: {
                true: 1,
                false: 0
            }
        });
    }
});
