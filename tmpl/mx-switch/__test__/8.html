以下示例： 切换成开始时，切换成加载中状态，4s后切换为可操作状态
HTML Code
{{!text1}}  
<mx-switch 
    state="{{=state}}" 
    loading="{{=loading}}"
    mx-change="change()"/>
JS Code
{{!text2}}  
let Magix = require('magix');
module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        this.updater.digest({
            state: false,
            loading: false
        });
    },
    'change<change>'(e) {
        let state = e.state;
        this.updater.digest({
            state,
            loading: state,
        })

        setTimeout(() => {
            this.updater.digest({
                loading: false,
            })
        }, 4000);
    }
});
