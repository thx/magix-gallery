let Magix = require('magix');

module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        this.updater.digest({
            state: true
        });
    },
    'toggle<change>' (event) {
        // event.state 当前开还是关
        // true 开，false关
        this.updater.digest({
            state: event.state
        });
    },
    'change<click>'(event){
        let state = this.updater.get('state');
        this.updater.digest({
            state: !state
        });
}
});
