以下示例：
动态刷新列表，历史已选中的传入完整的items:array[object]，历史选中不一定在可选列表中
输入关键词为空是存在默认可选list；用户输入关键词，动态更新提示数据
请求数据有延时，可加loading
当前选中： {{each items as item}} {{=item.value}}， {{/each}}
HTML Code
{{!text1}}  
<mx-taginput class="w280"  
    id="{{=viewId}}_taginput"
    placeholder="请选择需要的值"
    dynamic-list="true"
    items="{{@items}}"
    mx-show="update()"
    mx-change="change()"/>
JS Code
{{!text2}}  
let Magix = require('magix');
let Vframe = Magix.Vframe;
module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        this.updater.digest({
            items: [{
                text: 'one',
                value: 1
            }, {
                text: 'two',
                value: 2
            }]
        });
    },
    'update<show>'(e) {
        let keyword = e.keyword;
        let instance = Vframe.get(this.id + '_taginput');
        if (!keyword) {
            // 搜索关键词为空时，直接更新list
            let list = [1, 2, 3].map(v => {
                return {
                    text: `默认选项_${v}`,
                    value: `def_${v}`,
                };
            })
            instance.invoke('update', [list]);
        } else {
            // 加loading效果，处理延时请求
            instance.invoke('showLoading');
            setTimeout(() => {
                let list = [1, 2, 3].map(v => {
                    return {
                        text: Magix.guid('dynamic_'),
                        value: Magix.guid('dynamic_'),
                    };
                })
                instance.invoke('update', [list]);
                instance.invoke('hideLoading');
            }, 1000);
        }
    },
    'change<change>'(e) {
        this.updater.digest({
            items: e.items
        })
    }
});
