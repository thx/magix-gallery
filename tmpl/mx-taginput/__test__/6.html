JS Code
{{!text2}}  
let Magix = require('magix');

module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        let list = [];
        for (let index = 0; index < 5; index++) {
            list.push({
                text: 'test' + index,
                value: index
            })
        }
        this.updater.digest({
            viewId: this.id,
            list
        });
    },
    'select<change>'(e) {
        // e.selected：string，逗号分隔，选中value
        // e.items：array，选中完成对象
        this.updater.digest({
            selected: e.selected
        });
    }
});
