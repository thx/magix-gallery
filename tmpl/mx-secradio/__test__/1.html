以下示例：
mx-change实时获取选中值
支持展开收起功能
父节点，子节点都有前缀
当前选择：{{=selected}}
HTML Code
{{!text1}}  
<mx-secradio 
    list="{{@list}}"
    selected="{{=selected}}"
    parent-prefix="计划"
    prefix="单元"
    mx-change="select()"/>
JS Code
{{!text2}}  
let Magix = require('magix');
let $ = require('$');

module.exports = Magix.View.extend({
    tmpl: '@html.html',
    render() {
        let list = [{
            text: '单品推广计划1',
            subs: [{
                value: 11,
                text: '单品推广计划1_单元1'
            }, {
                value: 12,
                text: '单品推广计划1_单元2'
            }]
        }, {
            text: '单品推广计划2',
            subs: [{
                value: 21,
                text: '单品推广计划2_单元1'
            }, {
                value: 22,
                text: '单品推广计划2_单元2'
            }]
        }]
        this.updater.digest({
            list,
            selected: 21
        });
    },
    'select<change>'(event){
        // $(event.eventTarget) 当前组件
        this.updater.digest({
            selected: $(event.eventTarget).val()
        });
    }
});
