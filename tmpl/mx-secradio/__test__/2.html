let Magix = require('magix');
let $ = require('$');

module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        this.updater.digest({
            viewId: this.id,
            selected: ''
        });
    },
    'get<click>'(event){
        let comp = $('#' + this.id + '_comp');
        this.updater.digest({
            selected: comp.val()
        })
    }
});
