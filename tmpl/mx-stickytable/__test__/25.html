let Magix = require('magix');
module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        let fields = [];
        for (let i = 0; i < 5; i++) {
            fields.push({
                value: i,
                width: 180,
                minWidth: 100,
                maxWidth: 600
            })
        }
        this.updater.digest({
            fields,
            line: 4,
            index: 1
        });
    },
    'drag<dragfinish>'(e) {
        let items = e.items;
        let { fields } = this.updater.get();
        fields.forEach(field => {
            field.width = items[field.value] || field.width;
        })
        this.updater.digest({
            fields
        })
    }
});
