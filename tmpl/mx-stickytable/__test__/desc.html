1. mx-stickytable为dom增强型组件，涉及操作dom，建议将渲染表格涉及的参数直接挂载在组件上，确保每次digest更新时，数据能和展现保持一致
<mx-stickytable list="{{@list}}" fields="{{@fields}}">
    <table>
        <thead>
            <tr>
                {{each fields as field}}
                <th width="100">{{=field.text}}</th>
                {{/each}}
            </tr>
        </thead>
        <tbody>
            {{each list as item}}
            <tr>
                {{each fields as field}}
                <td>{{=item[field.value]}}</td>
                {{/each}}
            </tr>
            {{/each}}
        </tbody>
    </table>
</mx-stickytable>
