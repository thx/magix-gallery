import Magix from 'magix';
export default Magix.View.extend({
    tmpl: '@index.html',
    render() {
        this.updater.digest({
            expand: true
        });
    },
    'toggle<click>'(e) {
        // 如果有动画，需要抛出navslidend通知变更
        let { expand } = this.updater.get();
        this.updater.digest({
            expand: !expand,
        })
    }
});
