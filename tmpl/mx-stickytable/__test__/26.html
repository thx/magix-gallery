let Magix = require('magix');
module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        let fields = [{
            value: 1,
            width: 200
        }, {
            value: 2,
            width: 400
        }, {
            value: 3,
            width: 200
        }, {
            value: 4,
            width: 400
        }];

        this.updater.digest({
            fields
        });
    },
    'drag<dragfinish>'(e) {
        let items = e.items;
        let { fields } = this.updater.get();
        fields.forEach(field => {
            field.width = items[field.value] || field.width;
        })
        this.updater.digest({
            fields
        })
    }
});
