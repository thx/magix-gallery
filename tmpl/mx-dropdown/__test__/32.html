自定义提交校验函数，最多选择两个
HTML Code
{{!text1}}  
<mx-dropdown.multiple class="w280"  
    name="数字"
    list="{{@[1,2,3,4,5,6]}}"
    submit-checker="{{@checker}}"/>
JS Code
{{!text2}}  
let Magix = require('magix');

module.exports = Magix.View.extend({
    tmpl: '@32.html',
    render() {
        let that = this;
        that.updater.digest({
            checker: (selected) => {
                return new Promise(resolve => {
                    resolve((selected.length > 2) ? '最多选择两个' : '');
                })
            }
        });
    }
});
