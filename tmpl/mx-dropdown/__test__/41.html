let Magix = require('magix');
module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        this.updater.digest({
            selected: 2,
            list: [{
                value: 1,
                text: '单选1'
            }, {
                value: 3,
                text: '多选1',
                multiple: true
            }, {
                value: 4,
                text: '多选2',
                multiple: true
            }, {
                value: 5,
                text: '多选3',
                multiple: true
            }, {
                value: 2,
                text: '单选2'
            }]
        });
    }
});
