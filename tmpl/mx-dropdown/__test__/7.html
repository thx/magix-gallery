以下示例：
带分组的只能用mx-dropdown.item节点实现
不需要全选功能，配置need-all="false"
组上面需要全选功能在mx-dropdown.item上配置all="true"
展开选择浮层时会修正浮层到可视范围之内
第一组 {{each first as i}} {{=i}} {{/each}} 第二组 {{each second as i}} {{=i}} {{/each}}
HTML Code
{{!text1}}  
<mx-dropdown.multiple class="w200 mr20 fl"
    need-all="false"
    searchbox="true"
    name="数字"
    empty-text="请选择数字"
    selected="{{=selected}}">
    <mx-dropdown.item group="true" all="true">
        第一组
    </mx-dropdown.item>
    {{each first as i}}
    <mx-dropdown.item value="{{=i}}">{{=i}}</mx-dropdown.item>
    {{/each}}
    <mx-dropdown.item group="true" all="true">
        第二组
    </mx-dropdown.item>
    {{each second as i}}
    <mx-dropdown.item value="{{=i}}">{{=i}}</mx-dropdown.item>
    {{/each}}
</mx-dropdown.multiple>

<mx-btn brand="true" content="修改数据"
    mx-click="changeData()" />
JS Code
{{!text2}}  
let Magix = require('magix');
let $ = require('$');

module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        let that = this;

        let first = [],
            second = [];
        for (var i = 0; i < 20; i++) {
            first.push('1' + i);
            second.push('2' + i);
        }
        that.updater.digest({
            first,
            second,
            selected: [11, 12]
        });
    },
    'changeData<click>' (e) {
        let first = [],
            second = [];
        for (var i = 0; i < 20; i++) {
            first.push('3' + i);
            second.push('4' + i);
        }

        this.updater.digest({
            first,
            second,
            selected: [31]
        });
    }
});
