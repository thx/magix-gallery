JS Code
{{!text2}}  
let Magix = require('magix');

module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        this.updater.digest({
            selected: 2,
            list: [{
                value: 1,
                text: 'test'
            }, {
                value: 2,
                text: 'TestABC'
            }, {
                value: 3,
                text: 'Another'
            }]
        });
    },
    'change<change>'(event) {
        // event.value：string
        // event.text：string
        // event.selected：string
        this.updater.digest({
            selected: event.selected
        })
    }
});
