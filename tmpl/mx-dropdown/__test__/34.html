以下示例：
配置submit-checker自定义提交前校验函数
submit-checker需要返回一个promise
HTML Code
{{!text1}}  
<mx-dropdown.bd class="w280"
    multiple="true"
    list="{{@[1,2,3,4,5,6]}}" 
    selected="{{=selected}}" 
    submit-checker="{{@checker}}"/>
JS Code
{{!text2}}  
let Magix = require('magix');
module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        this.updater.digest({
            selected: [1, 2],
            checker: (items) => {
                // items: 当前选中数据对象数组
                return new Promise(resolve => {
                    resolve((items.length > 2) ? '最多选中两个' : '');
                })
            }
        });
    }
});
