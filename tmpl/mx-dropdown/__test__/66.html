点击展开后，异步获取列表展示数据
当前选中：{{=selected}}
HTML Code
{{!text1}}  
<mx-dropdown.bd class="w120"
    list="{{@list}}" 
    selected="{{=selected}}"
    fetch-loading="{{=loading}}"
    mx-focusin="getData()" 
    mx-change="change()" />
JS Code
{{!text2}}  
import Magix from 'magix';
export default Magix.View.extend({
    tmpl: '@index.html',
    render() {
        this.updater.digest({
            loading: true,
            list: [],
            selected: '',
        });
    },
    'change<change>'(event) {
        this.updater.digest({
            selected: event.selected,
        })
    },
    'getData<focusin>'() {
        setTimeout(() => {
            let { list: oldList } = this.updater.get();
            if (!oldList?.length) {
                let list = [];
                for (let i = 1; i < 10; i++) {
                    list.push({
                        text: Magix.guid('text_'),
                        value: i,
                    })
                };

                this.updater.digest({
                    loading: false,
                    list,
                    selected: '',  // 回置选中项
                })
            }
        }, 2000);
    }
});
