当前选中名： {{!selectedText}} 当前选中值： {{=selectedValue}}

JS Code

let Magix = require('magix');

module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        let list = [{
            value: 1,
            text: '<i class="mc-iconfont mr5">&#xe670;</i>测试Logo'
        }, {
            value: 2,
            text: 'logo'
        }, {
            value: 3,
            text: '123456'
        }];
        this.updater.digest({
            list: list,
            selectedValue: list[1].value
        });
    },
    'select<change>' (e) {
        // e.value 当前选中值
        // e.text 当前选中文案
        // e.item 当前选中完整对象
        // e.selected = e.value 
        // e.keyword 当前搜索关键词
    }
});
