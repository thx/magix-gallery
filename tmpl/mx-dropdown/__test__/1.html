带分组的只能用mx-dropdown.item节点实现
点击展开下拉选项
第一组 {{each first as i}} {{=i}} {{/each}} 第二组 {{each second as i}} {{=i}} {{/each}}
HTML Code
{{!text1}}  
<mx-dropdown class="w200 mr20"
    searchbox="true" 
    selected="{{=selected}}">
    <mx-dropdown.item group="true">第一组</mx-dropdown.item>
    {{each first as i}}
    <mx-dropdown.item value="{{=i}}">{{=i}}</mx-dropdown.item>
    {{/each}}
    <mx-dropdown.item group="true">第二组</mx-dropdown.item>
    {{each second as i}}
    <mx-dropdown.item value="{{=i}}">{{=i}}</mx-dropdown.item>
    {{/each}}
</mx-dropdown>

<mx-btn mode="primary" content="修改数据"
    mx-click="changeData()" />
JS Code
{{!text2}}  
let Magix = require('magix');

module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        let that = this;

        let first = [],
            second = [];
        for (var i = 0; i < 2; i++) {
            first.push('1' + i);
            second.push('2' + i);
        }
        that.updater.digest({
            first,
            second,
            selected: 11
        });
    },
    'changeData<click>' (e) {
        let first = [],
            second = [];
        for (var i = 0; i < 2; i++) {
            first.push('3' + i);
            second.push('4' + i);
        }

        this.updater.digest({
            first,
            second,
            selected: 31
        });
    }
});
