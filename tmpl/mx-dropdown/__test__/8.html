说明：
选项支持html片段
选中项传入支持传入数组或者逗号分隔
以下两个下拉框选中项保持一致
当前选中值： {{=selectedArr}}
HTML Code
{{!text1}}  
<mx-dropdown.multiple class="w200"
    list="{{@list}}"
    selected="{{@selectedArr}}"
    mx-change="select()"/>

<mx-dropdown.multiple class="w200"
    list="{{@list}}"
    selected="{{@selectedStr}}"
    mx-change="select()"/>
JS Code
{{!text2}}  
let Magix = require('magix');

module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        let list = [];
        for (let index = 0; index < 6; index++) {
            list.push({
                text: (index == 0 ? '<i class="mc-iconfont">&#xe670;</i>' : '') + '多选' + (index + 1),
                value: index + 1
            })
        }
        this.updater.digest({
            list,
            selectedArr: [1, 2, 3],
            selectedStr: '1,2,3'
        });
    },
    
    'select<change>'(e) {
        // e.values [array] 当前选中值
        // e.texts [array] 当前选中文案
        let values = e.values;
        this.updater.digest({
            selectedArr: values,
            selectedStr: values.join(',')
        });
    }
});
