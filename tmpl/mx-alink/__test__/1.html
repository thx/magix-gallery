禁用编辑 hover禁用原因
{{!text2}}  
<mx-alink content="编辑" mx-click="alert()"/>

<!-- 禁用：禁用状态下点击事件无响应 -->
<mx-alink disabled="true" mx-click="alert()">禁用编辑</mx-alink>

<!-- 禁用 + 禁用原因：禁用状态下点击事件无响应-->
<mx-alink disabled="true" disabled-tip="禁用提示信息" mx-click="alert()">hover禁用原因</mx-alink>
以下示例：
显示外链
{{!text3}}  
<!-- 外链 -->
<mx-alink icon="link" content="外链" link-href="https://www.taobao.com/" />

<!-- 外链 + 禁用 -->
<mx-alink disabled="true" icon="link" content="外链禁用" link-href="https://www.taobao.com/" />

<!-- 外链 + 禁用 + 禁用原因 -->
<mx-alink disabled="true" disabled-tip="禁用原因" icon="link" content="hover禁用原因" link-href="https://www.taobao.com/" />
