以下示例：
显示的数值范围为0 ~ 100，修正非法值
展示精度与传入数值保持一致，最多两位
支持配置提示文案位置在上下左右，默认为百分比
HTML Code
{{!text1}}  
<!-- 小于0显示为0，文案默认上方跟随进度显示 -->
<mx-effects.progress 
    num="-20.3" />

<!-- 最多显示两位小数，上跟随 -->
<mx-effects.progress 
    num="22.346" />

<!-- 文案在左侧 -->
<mx-effects.progress 
    num="43.2" 
    text-placement="left" />

<!-- 文案在右侧 -->
<mx-effects.progress 
    num="65.23" 
    text-placement="right" />

<!-- 文案在下方 -->
<mx-effects.progress 
    num="80" 
    text-placement="bottom" />

<!-- 超出100修正为100 -->
<mx-effects.progress 
    num="101.2" 
    text-placement="bottom" />
