let Magix = require('magix');
let $ = require('$');

module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        let list = [];
        for (let j = 0; j < 6; j++) {
            list.push({
                title: `消费者洞察报告${j}`,
                tip: '美妆行业新人群、新趋势、新场景机会，后风险时代的红海突围和蓝海挖掘，如何突破？',
                img: 'https://img.alicdn.com/tfs/TB1x5bkx.T1gK0jSZFrXXcNCXXa-2066-864.png',
                btns: [{
                    btn: '预览',
                    link: '', // 外链地址，不配置不跳转
                    outer: false // 新开页打开还是本页打开
                }, {
                    btn: '下载'
                }]
            })
        }

        this.updater.digest({
            list,
            viewId: this.id
        });
    },
    
    'select<select>'(e) {
        // item：当前卡片完成数据
        // btn：当前点击按钮完整数据
        let { item, btn } = e;
        this.updater.digest({
            selectedItem: item.title,
            selectedBtn: btn.btn
        })
    }
});
