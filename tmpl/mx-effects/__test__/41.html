let Magix = require('magix');

module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        let list = [];
        for (let j = 0; j < 2; j++) {
            list.push({
                title: `卡片${j}`,
                tip: '报告洞察每个环节的消费者触达效率，同时提供了相应的优化建议，有效挖掘市场机会。',
                img: 'https://img.alicdn.com/tfs/TB1V3iadmf2gK0jSZFPXXXsopXa-692-400.png'
            })
        }

        this.updater.digest({
            list,
            selected: {},
            viewId: this.id
        });
    },
    'select<select>'(e) {
        this.updater.digest({
            selected: e.item
        })
    }
});
