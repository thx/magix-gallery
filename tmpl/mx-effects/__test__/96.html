export default Magix.View.extend({
    tmpl: '@index.html',
    render() {
        let list = [{
            text: '5 ~ 7',
            value: 1,
            title: '点击量'
        }, {
            text: '17个',
            value: 2,
            title: '计划',
            subs: [{
                text: '计划名称1',
                value: 21,
            }, {
                text: '计划名称2',
                value: 22,
            }, {
                text: '计划名称3',
                value: 23,
            }]
        }, {
            text: '8 ~ 18',
            value: 3,
            title: '展现量'
        }, {
            text: '超长提示文案超长提示文案超长提示文案超长提示文案',
            value: 4,
            title: '展示标题'
        }];
        this.updater.digest({
            list,
        });
    },
    'delete<delete>'(e) {
        // list 当前剩余完整列表
        // item 当前移除的对象（一级或者二级）
        let { list, item } = e;
        this.updater.digest({
            list: e.list,
        })
    }
});
