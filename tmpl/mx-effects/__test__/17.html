以下示例：
hover标签打开提示信息小浮层
支持配置tip-view自定义提示信息view，tip-data传入初始化数据
HTML Code
{{!text1}}  
<mx-effects.icon type="error" content="hover提示" tip="提示信息"/>
    
<mx-effects.icon mode="hollow" type="error"  content="hover提示" tip="提示信息"/>

<mx-effects.icon mode="hollow" type="error"  content="hover提示" tip="右上对齐" tip-placement="right" tip-align="top"/>

<mx-effects.icon mode="hollow"  type="error"  content="自定义view提示"
    tip-view="@./custom" tip-width="400"
    tip-data="{{@{
        list: [{
            name: '消耗',
            key: 'charge',
            type: 'double'
        }, {
            name: '展现量',
            key: 'adPv',
            type: 'integer'
        }, {
            name: '点击量',
            key: 'click',
            type: 'integer'
        }, {
            name: '点击率',
            key: 'ctr',
            type: 'percent'
        }]
    }}}" />
