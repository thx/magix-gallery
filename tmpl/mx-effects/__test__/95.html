import Magix from 'magix';

export default Magix.View.extend({
    tmpl: '@index.html',
    render() {
        let list = [];
        for (let i = 0; i < 10; i++) {
            list.push({
                text: `选项${i}`,
                value: i,
            })
        }
        this.updater.digest({
            list,
        });
    },
    'delete<delete>'(e) {
        // list 当前剩余完整列表
        // item 当前移除的对象
        let { list, item } = e;
        this.updater.digest({
            list,
        })
    }
});
