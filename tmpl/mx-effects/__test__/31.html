let Magix = require('magix');

module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        this.updater.digest({
            index: 1
        });
    },
    'change<click>'(event) {
        let { index } = this.updater.get();
        this.updater.digest({
            index: index + 1
        });
    }
});
