以下示例：
实心样式icon，color配置背景色，文案默认白色，可配置color-text自定义文案颜色
以下示例：
空心样式icon，color配置边框颜色，默认边框文案同色，可配置color-text自定义文案颜色
以下示例：
透明度icon，背景色color透明度0.1，文案默认color，可配置color-text自定义文案颜色
HTML Code
{{!text1}}  
<!-- 实心样式icon，color配置背景色，文案默认白色 -->
<mx-effects.icon mode="solid" color="#51a300"  content="测试"/>

<!-- 实心样式icon，color配置背景色，color-text配置自定义文案颜色-->
<mx-effects.icon mode="solid" color="#303a58"  color-text="#ffb400" content="测试"/>

<!-- 空心样式icon，color配置边框颜色，默认边框文案同色 -->
<mx-effects.icon mode="hollow" color="#51a300" content="测试"/>

<!-- 空心样式icon，color配置边框颜色，color-text配置自定义文案颜色 -->
<mx-effects.icon mode="hollow" color="#999999" color-text="#ffb400"content="测试"/>

<!-- 透明度icon，背景色color透明度0.1，文案默认color -->
<mx-effects.icon mode="opacity" color="#51a300" content="测试"/>

<!-- 透明度icon，背景色color透明度0.1，color-text配置自定义文案颜色 -->
<mx-effects.icon mode="opacity" color="#51a300" color-text="#ffb400"content="测试"/>
