以下示例：
配置reload="true"带重刷icon，点击处罚事件reload
0-19红色，20-59黄色，60-99蓝色，100绿色
当前进度：
{{=num}}
HTML Code
{{!text1}}  
<mx-effects.progress 
    type="{{=(num < 20 ? 'circle-error' : (num < 60 ? 'circle-warn' : (num < 100 ? 'circle-stress' : 'circle-pass')))}}"
    num="{{=num}}"
    reload="{{=(num < 100)}}"
    prefix="{{=(num < 20 ? '上传故障中' : (num < 60 ? '上传中断' : (num < 100 ? '上传中' : '上传完成')))}}" 
    mx-reload="reload()"/>
JS Code
{{!text2}}  
let Magix = require('magix');
module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        this.updater.digest({
            num: 18,
        });
    },
    'reload<reload>'(e) {
        let { num } = this.updater.get();
        this.updater.digest({
            num: num + 30
        })
    }
});
