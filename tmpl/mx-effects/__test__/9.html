let Magix = require('magix');
let $ = require('$');

module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        this.updater.digest({
            num1: 20
        });
    },
    'add<click>'(e){
        let data = this.updater.get();
        this.updater.digest({
            num1: data.num1 + 10
        });
    }
});
