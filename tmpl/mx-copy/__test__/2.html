JS Code
{{!text3}}  
let Magix = require('magix');
module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        this.updater.digest({
            viewId: this.id
        });
    },
    'done<success>'(e) {
        this.updater.digest({
            success: true
        })
    }
});
