以下示例：
选择结束投放操作时（confirmTitle + confirmContent），会进行二次确认
确定之后继续操作，取消则无响应
当前选中项： {{=cur.text}}（{{=cur.value}}）
HTML Code
{{!text1}}  
<mx-status 
    opers="{{@opers}}" 
    selected="{{=cur.value}}"
    mx-change="change()"/>
JS Code
{{!text2}}  
let Magix = require('magix');

module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        let opers = [{
            value: 0,
            text: '暂停投放',
            icon: '<i class="mc-iconfont">&#xe734;</i>',
            color: 'var(--color-red)'
        }, {
            value: 9,
            text: '结束投放',
            confirmTitle: '系统提示',
            confirmContent: '确认结束投放吗？',
            icon: '<i class="mc-iconfont">&#xe6dd;</i>',
            color: '#666666'
        }]

        this.updater.digest({
            opers,
            cur: opers[1]
        });
    },
    'change<change>'(e){
        // e.status 完整的对象
        //      value
        //      text
        this.updater.digest({
            cur: e.status
        });
    }
});
