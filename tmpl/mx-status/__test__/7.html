提示 + 操作项
显示提示信息，hover显示操作项的状态
当前选中项： {{=cur.text}}（{{=cur.value}}）
HTML Code
{{!text1}}  
<mx-status.dropdown 
    info="{{@info}}"
    opers="{{@opers}}" 
    selected="{{=cur.value}}"
    mx-change="change()"/>
JS Code
{{!text2}}  
let Magix = require('magix');
module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        let opers = [{
            value: 1,
            text: '正在投放',
            icon: '<i class="mc-iconfont"></i>',
            color: '#51a300'
        }, {
            value: 9,
            text: '结束投放',
            confirmTitle: '系统提示',
            confirmContent: '确认结束投放吗？',
            icon: '<i class="mc-iconfont"></i>',
            color: '#999999'
        }];
        this.updater.digest({
            info: {
                value: -99,
                text: '故障',
                icon: '<i class="mc-iconfont"></i>',
                color: '#a40100'
            },
            opers,
            cur: opers[0]
        });
    },
    'change<change>'(e) {
        // e.status 完整的对象
        //      value
        //      text
        this.updater.digest({
            cur: e.status
        });
    }
});
