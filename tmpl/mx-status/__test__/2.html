module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        this.updater.digest({
            tipInfo: {
                value: 1,
                text: '正在投放',
                icon: '<i class="mc-iconfont">&#xe735;</i>',
                tip: '当前正在投放中<br/>换一行',
                color: 'var(--color-green)'
            },
            viewInfo: {
                value: 1,
                text: '正在投放',
                icon: '<i class="mc-iconfont">&#xe735;</i>',
                tipView: '@./tip',
                tipData: {
                    tag: '预算过低',
                    tip: '建议修改预算至10000（+15%），相比当前展现量+35%，点击量+20%'
                },
                color: 'var(--color-green)'
            }
        });
    }
});
