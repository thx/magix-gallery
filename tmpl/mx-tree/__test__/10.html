使用场景： 切换页码的时候保留历史选中项
当前选中单元（翻页时更新）： {{=selected}}
HTML Code
{{!text1}}  
<mx-tree 
    mode="radio"
    list="{{@list}}"
    selected="{{:selected{refresh:true}}}" />

<mx-pagination class="table-pager-wrapper"
    total="10" 
    size="5" 
    page="1"
    mini="true"
    jump="false"
    mx-change="changePager()"/>

let Magix = require('magix');
module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        let data = {
            1: [{
                value: 1,
                pValue: '',
                text: '计划1'
            }, {
                value: 2,
                pValue: '',
                text: '计划2'
            }, {
                value: 11,
                pValue: 1,
                text: '单元1-1'
            }, {
                value: 12,
                pValue: 1,
                text: '单元1-2'
            }, {
                value: 21,
                pValue: 2,
                text: '单元2-1'
            }, {
                value: 22,
                pValue: 2,
                text: '单元2-2'
            }],
            2: [{
                value: 3,
                pValue: '',
                text: '计划3'
            }, {
                value: 4,
                pValue: '',
                text: '计划4'
            }, {
                value: 31,
                pValue: 3,
                text: '单元3-1'
            }, {
                value: 32,
                pValue: 3,
                text: '单元3-2'
            }, {
                value: 41,
                pValue: 4,
                text: '单元4-1'
            }, {
                value: 42,
                pValue: 4,
                text: '单元4-2'
            }]
        }
        this.updater.digest({
            data,
            list: data[1],
            selected: '',
        });
    },
    'changePager<change>'(e) {
        let { data } = this.updater.get();
        this.updater.digest({
            list: data[e.page],
        })
    }
});
