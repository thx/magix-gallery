以下示例：
选中值始终返回叶子节点
支持双向绑定
选中单元：
{{each bottomItems as item}} {{=item.text}}（{{=item.value}}）； {{/each}}
HTML Code
{{!text1}}  
<mx-tree 
    need-all="true"
    list="{{@list}}"
    bottom-values="{{@bottomValues}}"
    mx-change="change()"  />
JS Code
{{!text2}}  
let Magix = require('magix');
module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        let list = [{
            value: 1,
            pValue: '',
            text: '计划1'
        }, {
            value: 2,
            pValue: '',
            text: '计划2'
        }, {
            value: 11,
            pValue: 1,
            text: '单元1-1'
        }, {
            value: 12,
            pValue: 1,
            text: '单元1-2'
        }, {
            value: 21,
            pValue: 2,
            text: '单元2-1'
        }, {
            value: 22,
            pValue: 2,
            text: '单元2-2'
        }]

        this.updater.digest({
            list,
            bottomValues: [],
            bottomItems: []
        });
    },
    'change<change>'(e) {
        // e.bottomValues：已选中的最底层value列表，入参为bottom-values时返回
        // e.bottomItems：已选中的最底层完整对象，入参为bottom-values时返回
        this.updater.digest({
            bottomValues: e.bottomValues,
            bottomItems: e.bottomItems,
        })
    },
});
