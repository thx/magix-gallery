1. 文字预览：支持内容包含可执行脚本，以文字展示
2. 文字预览：可配置文字上的跳转链接click-url，可不配width和height，不配则width=600，height=auto，根据屏幕范围修正
文字预览
{{!text1}}  
<!-- 内容包含可执行脚本 -->
<mx-preview  
    border="true"
    type="text"
    url="<script>alert(1)</script>"
    width="200"
    height="200"
    max-width="100"
    max-height="100"/>

<!-- 文字上可配置跳转链接 -->
<mx-preview 
    border="true"
    type="text"
    url="文字链内容展示"
    click-url="//www.taobao.com/"
    max-width="100"
    max-height="100"/>
