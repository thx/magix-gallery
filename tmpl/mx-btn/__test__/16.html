以下示例：
配置arrow显示箭头，结合active使用激活态
HTML Code
{{!text1}}  
<mx-btn 
    mode="primary" 
    content="主要按钮" 
    arrow="true" 
    active="{{=active}}" 
    mx-click="toggle()"/>
HTML Code
{{!text2}}  
import Magix from 'magix';

export default Magix.View.extend({
    tmpl: '@index.html',
    render() {
        this.updater.digest({
            active: false,
        });
    },
    'toggle<click>'(e) {
        this.updater.digest({
            active: !this.updater.get('active'),
        });
    }
});
