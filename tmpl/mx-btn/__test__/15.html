以下示例：
禁用状态下点击事件无响应
禁用编辑 hover禁用原因
{{!text2}}  
<mx-btn.link icon="<i class='mx-iconfont'>&#xe64e;</i>" content="编辑" mx-click="alert()"/>

<!-- 禁用：禁用状态下点击事件无响应 -->
<mx-btn.link icon="<i class='mx-iconfont'>&#xe64e;</i>" disabled="true" mx-click="alert()">禁用编辑</mx-btn.link>

<!-- 禁用 + 禁用原因：禁用状态下点击事件无响应-->
<mx-btn.link icon="<i class='mx-iconfont'>&#xe64e;</i>" disabled="true" disabled-tip="禁用提示信息" mx-click="alert()">hover禁用原因</mx-btn.link>

显示外链

<!-- 外链 + 禁用 + 禁用原因 -->
<mx-btn.link disabled="true" disabled-tip="禁用原因" icon="link" content="hover禁用原因" link-href="https://www.taobao.com/" />
