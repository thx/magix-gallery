let Magix = require('magix');

module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        this.updater.digest({
            loading: false
        });
    },
    'change<click>'(e) {
        let { loading } = this.updater.get();
        this.updater.digest({
            loading: !loading
        });
    }
});
