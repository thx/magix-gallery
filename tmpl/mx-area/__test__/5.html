HTML Code

<mx-area id="{{=viewId}}_area"
    selected="{{@values}}"
    mx-change="change()"/>
JS Code
{{!text2}}  
let Magix = require('magix');
let $ = require('$');

module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        this.updater.digest({
            viewId: this.id,
            values: [1]
        });
    },
    'change<change>'(e) {
        // values: 当前选中项的id
        // selected: 当前选中项完整对象
        this.updater.digest({
            values: e.values
        })
    }
});
