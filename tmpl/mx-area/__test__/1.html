以下示例：
可以选择城市
每个省份和城市都有各自的id，若选择了省份，则获取选中值时只有省份的id（不包含该省份下所有城市id）
宽度由容器宽度决定，自适应
无需实时响应变更结果，一次性获取选中项
当前选中值：
{{each selected as item}}
{{=item}}
{{/each}}
HTML Code
{{!text1}}  
<mx-area id="{{=viewId}}_area"
    city="true"
    selected="{{@selected}}"/>
JS Code
{{!text2}}  
let Magix = require('magix');
let $ = require('$');

module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        this.updater.digest({
            viewId: this.id,
            selected: [1]
        });
    },
    'get<click>' (event) {
        let id = this.id + '_area'
        let selected = Magix.Vframe.get(id).invoke('val');
        this.updater.digest({
            selected: selected
        })
    }
});
