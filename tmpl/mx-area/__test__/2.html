let Magix = require('magix');
let $ = require('$');

module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        this.updater.digest({
            viewId: this.id,
            selected: [1]
        });
    },
    'get<click>' (event) {
        let id = this.id + '_area'
        let selected = Magix.Vframe.get(id).invoke('val');
        this.updater.digest({
            selected: selected
        })
    }
});
