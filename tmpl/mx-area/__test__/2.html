以下示例：
无需实时反馈选中结果，仅需要最终获取一次数据结果
当前选中值：
{{each values as item}}
{{=item}}
{{/each}}
HTML Code
{{!text1}}  
<mx-btn mode="primary" content="获取选中值" 
    mx-click="get()"/>

<mx-area id="{{=viewId}}_area"
    values="{{@values}}"/>
JS Code
{{!text2}}  
let Magix = require('magix');
module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        this.updater.digest({
            viewId: this.id,
            values: [1]
        });
    },
    'get<click>' (event) {
        let id = this.id + '_area'
        let values = Magix.Vframe.get(id).invoke('val');
        this.updater.digest({
            values,
        })
    }
});
