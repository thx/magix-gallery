以下示例：
打开全屏遮罩，2秒后自动关闭
默认黑色遮罩，文案显示“加载中...”
点击打开遮罩
HTML Code
{{!text1}}  
<a href="javascript:;" class="link-brand"
    mx-click="show()">点击打开遮罩</a>
JS Code
{{!text2}}  
let Magix = require('magix');
let Loading = require('@../../mx-loading/mask');
module.exports = Magix.View.extend({
    tmpl: '@index.html',
    mixins: [Loading],
    render() {
        this.updater.digest();
    },
    'show<click>'(e){
        let that = this;
        if (that.timer) {
            clearTimeout(that.timer);
        }

        that.showLoading();
        
        that.timer = setTimeout(() => {
            that.hideLoading();
        }, 4000);
    }
});
