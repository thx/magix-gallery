let Magix = require('magix');
let Loading = require('@../../mx-loading/mask');
module.exports = Magix.View.extend({
    tmpl: '@index.html',
    mixins: [Loading],
    render() {
        this.updater.digest();
    },
    'show<click>'(e){
        let that = this;
        if (that.timer) {
            clearTimeout(that.timer);
        }

        // 指定相对loading节点，支持多节点
        that.showLoading({
            sizzle: `[data-table^="${that.id}"]`,
        });
        
        that.timer = setTimeout(() => {
            that.hideLoading();
        }, 2000);
    }
});
