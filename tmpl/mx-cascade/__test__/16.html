子选项中包含重复值，组件需要根据其父节点重置value保证其唯一性
选中值使用items传递完整的父子关系对象，支持双向绑定
当前选中： {{each items as item index}} {{=item.code}}{{if index < items.length - 1}} => {{/if}} {{/each}}
HTML Code
{{!text1}}  
<mx-cascade class="w200"
    adc-list="{{@adcList}}" 
    repeat="true"
    items="{{:items{refresh:true}}}" />
JS Code
{{!text2}}  
import Magix from 'magix';
export default Magix.View.extend({
    tmpl: '@index.html',
    render() {
        this.updater.digest({
            items: [{
                name: '选项1',
                code: 1,
            }, {
                name: '选项1_1',
                code: 1,
            }],
            adcList: [{
                name: '选项1',
                code: 1,
                subComponentList: [{
                    name: '选项1_1',
                    code: 1,
                }, {
                    name: '选项1_2',
                    code: 2,
                }]
            }, {
                name: '选项2',
                code: 2,
                subComponentList: [{
                    name: '选项2_1',
                    code: 1,
                }, {
                    name: '选项2_2',
                    code: 2,
                }]
            }],
        });
    }
});
