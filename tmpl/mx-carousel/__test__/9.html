let Magix = require('magix');

module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        this.updater.digest({
            viewId: this.id,
            list: ['var(--color-brand)', 'var(--color-green)', 'var(--color-warn)']
        });
    },
    'next<click>'(e) {
        let carousel = Magix.Vframe.get(`${this.id}_carousel`);
        carousel.invoke('next');
    },
    'prev<click>'(e) {
        let carousel = Magix.Vframe.get(`${this.id}_carousel`);
        carousel.invoke('prev');
    }
});
