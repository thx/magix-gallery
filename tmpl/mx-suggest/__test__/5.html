以下示例：
传入固定的list，根据输入的关键词在当前list中模糊检索，不区分大小写
支持配置type在不同的字段中检索，默认all（同时在text和value中检索）
在value和text中搜索（默认，type=all）：
当前选中： {{=selected3}}
只在text中搜索（type=text）：
当前选中： {{=selected2}}
只在value中搜索（type=value）：
当前选中： {{=selected1}}
HTML Code
{{!text1}}  
<!-- 在value和text中搜索（默认） -->
<mx-suggest class="w200" 
    selected="{{:selected3{refresh:true}}}" 
    list="{{@list}}" 
    type="all"/>

<!-- 只在text中搜索 -->
<mx-suggest class="w200" 
    selected="{{:selected2{refresh:true}}}" 
    list="{{@list}}" 
    type="text"/>

<!-- 只在value中搜索 -->
<mx-suggest class="w200" 
    selected="{{:selected1{refresh:true}}}" 
    list="{{@list}}" 
    type="value"/>
JS Code
{{!text2}}  
let Magix = require('magix');
module.exports = Magix.View.extend({
    tmpl: '@index.html',
    mixins: [Form, Validator],
    render() {
        this.updater.digest({
            list: [{
                text: 'test1',
                value: 4
            }, {
                text: 'Test2',
                value: 3
            }, {
                text: 'hello3',
                value: 2
            }, {
                text: 'Hello4',
                value: 1
            }],
            selected1: '',
            selected2: '',
            selected3: '',
        });
    }
});
