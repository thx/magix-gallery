以下示例：
传入固定的list，根据输入的关键词在当前list中模糊检索，不区分大小写
有可选项时，回车默认选中第一个
当前选中： {{if selectedValue}} {{=selectedValue}}（{{=selectedText}}） {{/if}}
HTML Code
{{!text1}}  
<mx-suggest class="w200"
    placeholder="输入关键词搜索"
    list="{{@list}}"
    selected="{{=selectedValue}}"
    mx-change="suggest()" />
JS Code
{{!text2}}  
let Magix = require('magix');
module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        this.updater.digest({
            list: [{
                text: 'test1',
                value: 1
            }, {
                text: 'test12',
                value: 2
            }, {
                text: 'abc1',
                value: 3
            }, {
                text: 'abc2',
                value: 4
            }],
            selectedValue: '',
            selectedText: ''
        });
    },
    'suggest<change>'(e) {
        // e.item [object]
        //      value
        //      text
        // e.selected [item.value]
        let item = e.item;
        this.updater.digest({
            selectedValue: item.value,
            selectedText: item.text
        })
    }
});
