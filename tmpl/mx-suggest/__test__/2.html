回车默认选中第一个

JS Code
{{!text2}}  
let Magix = require('magix');

module.exports = Magix.View.extend({
    tmpl: '@index.html',
    render() {
        this.updater.digest({
            selectedValue: ''
        });
    },
    'suggest<suggest>'(e) {
        // e.item [object]
        //      value
        //      text
        // e.selected [item.value]
        this.updater.digest({
            selectedValue: e.selected
        })
    }
});
