以下示例：
type=xhr时，可配置form-data对象，传入额外参数
HTML Code
{{!text1}}  
<mx-uploader 
    type="xhr"
    form-data="{{@{
        csrfId: 166796141446304865837199293810134,
        bizCode: 'display'
    }}}"
    action="/creative/uploadImages.action" 
    method="POST"
    name="images"
    accept="image/jpeg,image/png,image/jpg"
    mx-error="uploadError()"
    mx-success="uploadSuccess()">
    <mx-btn class="w80" icon="<i class='mx-iconfont'>&#xe718;</i>" mode="primary"  content="上传"/>    
</mx-uploader>
