监听mx-start在上传前对数据进行预处理
HTML Code
{{!text1}}  
<mx-uploader 
    action="/creative/uploadImages.action?csrfId={{=csrfId}}" 
    method="POST"
    name="images"
    accept="image/jpeg,image/png,image/jpg"
    mx-start="uploadStart()"
    mx-error="uploadError()"
    mx-success="uploadSuccess()" >
    <mx-btn class="w80" icon="<i class='mx-iconfont'>&#xe718;</i>" mode="primary"  content="上传"/>    
</mx-uploader>

    render() {

        });
    },
    'uploadStart<start>'(e) {
        // e.files
        // 异步处理
        e.result = new Promise((resolve, reject) => {
            // do something
            // resolve 继续提交
            // reject 中断提交
            resolve();
        })
    },
    'uploadSuccess<success>'(e) {
        // e.files
        // e.response
    },
    'uploadError<error>'(e) {
        // e.error
    }
});
