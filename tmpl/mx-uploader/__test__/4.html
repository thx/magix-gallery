以下示例：
分片上传，详细说明见 https://yuque.antfin-inc.com/fe-p2/xdsgi2/qv4f05
需要配置type=oss和 oss-action（获取token接口），其他和普通上传无差别
HTML Code
{{!text1}}  
<mx-uploader 
    type="oss"
    oss-action="/api/oss/getSts.json?token={{=token}}"
    action="/api/upload.json" 
    method="POST"
    name="fileName"
    accept="video/*"
    mx-start="uploadVideoStart()"
    mx-progress="uploadVideoProgress()"
    mx-error="uploadVideoError()"
    mx-success="uploadVideoSuccess()">
    <mx-btn class="w80" icon="<i class='mx-iconfont'>&#xe718;</i>" mode="primary"  content="上传"/>    
</mx-uploader>
